# usar la mejor distribución encontrada por fitter
best_dist_name = list(f.get_best(method='sumsquare_error').keys())[0]
params = f.fitted_param[best_dist_name]
dist = getattr(stats, best_dist_name)

# frecuencias observadas
bin_edges = np.histogram_bin_edges(data_feature, bins='auto')
observed_freq, bin_edges = np.histogram(data_feature, bins=bin_edges)
n = observed_freq.sum()

# frecuencias esperadas usando probabilidad por bin (CDF)
cdf_vals = dist.cdf(bin_edges, *params)
bin_probs = np.diff(cdf_vals)
expected_freq = n * bin_probs

# evitar bins inválidos (esperado 0) y normalizar suma
valid = expected_freq > 0
obs = observed_freq[valid]
exp = expected_freq[valid]
exp = exp * (obs.sum() / exp.sum())

# chi-cuadrado (ajustando por parámetros estimados)
chi2_stat, p_value = stats.chisquare(f_obs=obs, f_exp=exp, ddof=len(params))

print(f"Best distribution: {best_dist_name}")
print(f"Chi-square statistic: {chi2_stat:.4f}")
print(f"p-value: {p_value:.6f}")
print("Conclusion:", "No se rechaza H0 (buen ajuste)" if p_value > 0.05 else "Se rechaza H0 (mal ajuste)")
